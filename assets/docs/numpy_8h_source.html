73 inline int get_typenum(NanoString ns) {
74  if (ns == ns_bool) return 0;
75  if (ns == ns_int8) return 1;
76  if (ns == ns_uint8) return 2;
77  if (ns == ns_int16) return 3;
78  if (ns == ns_uint16) return 4;
79  if (ns == ns_int32 || ns == ns_int) return 5;
80  if (ns == ns_uint32) return 6;
81  if (ns == ns_int64) return 7;
82  if (ns == ns_uint64) return 8;
83  if (ns == ns_float32 || ns == ns_float) return 11;
84  if (ns == ns_float64 || ns == ns_double) return 12;
85  LOGf << ns;
86  return -1;
87 }

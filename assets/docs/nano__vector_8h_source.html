49  static inline int get_nbits(int64 v) {
50  // return 1bit if v in [-1,0]
51  // return 2bit if v in [-2,1,0,1]
52  if (v<0) v = ~v;
53  return 65 - lzcnt(v);
54  }

99  inline int64 at(int i) const {
100  if (i<0) i+= size();

103  int next_offset = get_offset(i);

110  inline NanoVector slice(Slice slice) {
111  if (slice.start<0) slice.start += size();
112  if (slice.stop<0) slice.stop += size();
113  ASSERT(slice.start>=0 && slice.stop>=0 && slice.start<size() && slice.stop<size())
114  << "slice overflow:" << slice.start << slice.stop << slice.step;
115  NanoVector v;
116  if (slice.step>0) {
117  for (int i=slice.start; i<slice.stop; i+=slice.step)
118  v.push_back(this->operator[](i));
119  } else {
120  for (int i=slice.start; i>=slice.stop; i+=slice.step)
121  v.push_back(this->operator[](i));
122  }
123  return v;
124  }

176  inline void reserve(int64 max_value, int n) {
177  auto nbits = get_nbits(max_value);
178  ASSERT(n<=10 && nbits*n<=64) << n << nbits;

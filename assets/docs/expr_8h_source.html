12 enum Flags {
13  // is op or is a symbol
14  _unary_op = 1<<7,
15  _binary_op = 1,
16  _ternary_op = 1<<8,

19  _call = 1<<1,

155 template<class... Args>
156 unique_ptr<Expr> make_op(const string& str, Args&&... args) {
157  vector<unique_ptr<Expr>> children;
158  children.reserve(sizeof...(args));
159  auto f = [&](unique_ptr<Expr>& c) { children.emplace_back(move(c)); };
160  // Brace-enclosed initializers
161  int dummy[] = {(f(args), 0)...};
162  (void)dummy;
163  return make(str, move(children));
164 }

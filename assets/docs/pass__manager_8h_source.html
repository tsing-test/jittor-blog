36 template <class T>
37 void PassManager::run_pass() {
38  auto pass = std::make_unique<T>();
39  if (!check(pass.get())) {
40  LOGvvv << "exclude pass" << pass->name;
41  return;
42  }
43  LOGvvv << "run pass" << pass->name;
44  pass->init(this);
45  pass->run();
46  LOGvvvv << "Kernel IR after pass" << pass->name << ":\n"
47  << main_ir->to_string(0, true);
48  pass_map.emplace(pass->name, pass.get());
49  finished_passes.push_back(move(pass));
50 }

52 template <class T>
53 T* PassManager::get_pass(const string& name) {
54  auto iter = pass_map.find(name);
55  if (iter == pass_map.end()) return nullptr;

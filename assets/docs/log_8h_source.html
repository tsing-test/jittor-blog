43 static void get_color(char level, int verbose, const char*& color_begin, const char*& color_end) {
44  if (level == 'i') {
45  if (verbose == 0) color_begin = "\033[38;5;2m"; else
46  if (verbose < 10) color_begin = "\033[38;5;250m"; else
47  if (verbose < 100) color_begin = "\033[38;5;244m"; else
48  if (verbose < 1000) color_begin = "\033[38;5;238m";
49  else color_begin = "\033[38;5;232m";
50  } else if (level == 'w')
51  color_begin = yellow;
52  else if (level == 'e')
53  color_begin = red;
54  else // level == 'f'
55  color_begin = red;
56  color_end = "\033[m";
57 }

124 template<class T> T get_from_env(const char* name,const T& _default) {
125  auto s = getenv(name);
126  if (s == NULL) return _default;
127  std::istringstream is(s);
128  T env;
129  if (is >> env) {
130  is.peek();
131  if (!is) {
132  return env;
133  }
134  }
135  LOGw << "Load" << name << "from env(" << s << ") failed, use default" << _default;
136  return _default;
137 }

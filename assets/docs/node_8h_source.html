15 struct NodeFlags {

18  enum Flags {

32  // bit0: support cpu
33  _cpu=_n+0,
34  // bit1: support cuda
35  _cuda=_n+1,
36  // bit2: forward op
37  _forwarded=_n+2,
38  // bit3: vary shape op
39  _vary_shape=_n+3,
40  // bit4~5: op type
41  _op_type=_n+4, _op_type_nbits=2,
42  };

44  inline void set(Flags f, int a=1, int nbits=1) {
45  nf_t mask = (((1u<<nbits)-1)<<f);
46  flags = (flags & ~mask) | ((a<<f)&mask);
47  }
48 
49  inline nf_t get(Flags f, int nbits=1) const {
50  return (flags>>f) & ((1u<<nbits)-1);
51  }

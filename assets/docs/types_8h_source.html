209  struct Iter {

211  Iter(typename list<To>::iterator iter)
212  : iter(iter), next(std::next(iter)) {}
213  T operator*() { return iter->operator T(); }
214  Iter& operator++() { iter = next++; return *this; }
215  Iter operator++(int) { auto tmp = *this; ++(*this); return tmp; }
216  bool operator!=(Iter& other) { return iter != other.iter; }
217  };
218  Iter begin() const { return Iter(ptr->begin()); }
219  Iter end() const { return Iter(ptr->end()); }
220  size_t size() { return ptr->size(); }
221  T front() { return ptr->front().operator T(); }
222  T back() { return ptr->back().operator T(); }
